@startuml

class Item{
    Vec3 position
    volume() : int
}

class Vec3

abstract Packer {
    setLimits()
    setPolicy()
    intersects(Item b, Item b) : bool
    intersectsAny(Item toPack) : bool
    fitsWithinLimits(Item toPack) : bool
    volume() : int
    usedVolume() : int
    {abstract} pack(Item toPack)
}

class GreedyPacker {
    pack(Item toPack)
}

interface IRotationPolicy {
    {abstract} rotateBin(Item toPack, Vec3 binSize)
}

class RP_largestFaceUp
class RP_minLeftoverSlack
class RP_tryFirstFitting

Packer <|--GreedyPacker
Packer *-right- IRotationPolicy
IRotationPolicy <|-- RP_largestFaceUp
IRotationPolicy <|-- RP_minLeftoverSlack
IRotationPolicy <|-- RP_tryFirstFitting

@enduml
